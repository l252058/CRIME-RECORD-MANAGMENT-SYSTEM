bool loginUser(string email, string password)
{
    ifstream file("users.txt");

    if (!file.is_open())
    {
        return false;
    }

    string name, fileEmail, filePassword, separator;

    while (getline(file, name) && getline(file, fileEmail) && getline(file, filePassword) && getline(file, separator))
    {
        if (fileEmail == email && filePassword == password)
        {
            file.close();
            return true;
        }
    }

    file.close();
    return false;
}
